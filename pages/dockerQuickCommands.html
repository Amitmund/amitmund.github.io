Docker Quick Commands


# Docker Run example
docker run --name web_server -d -p 8080:80 -v someDirectory/public_html:/usr/share/nginx/html nginx


# Docker Run another example
docker run -it --name oracleClient -v ~/Downloads/oracle_client:/root ubuntu /bin/bash


# To remove stopped docker instances
docker rm $(docker ps -aq)


# To create a image from a running instance
docker commit container_id container_name_amund:latest


# From docker image to tar file
https://docs.docker.com/engine/reference/commandline/save/

docker save busybox > busybox.tar
or
docker save --output busybox.tar busybox

docker save container_name_amund:latest > container_name_amund.tar


# From docker image to tar.gz
docker save myimage:latest | gzip > myimage_latest.tar.gz


# from tar file to image:
docker load --input fedora.tar



# copying from instance to local
docker cp containerId:/file/path/within/container /host/path/target

# copying from local to instance
docker cp foo.txt mycontainer:/foo.txt

Docker quick commands


# Containers
Use docker container my_command

create — Create a container from an image.
start — Start an existing container.
run — Create a new container and start it.
ls — List running containers.
inspect — See lots of info about a container.
logs — Print logs.
stop — Gracefully stop running container.
kill —Stop main process in container abruptly.
rm— Delete a stopped container.


# Images
Use docker image my_command
build — Build an image.
push — Push an image to a remote registry.
ls — List images.
history — See intermediate image info
inspect — See lots of info about an image, including the layers.
rm — Delete an image.


# Misc
docker version — List info about your Docker Client and Server versions.
docker login — Log in to a Docker registry.
docker system prune — Delete all unused containers, unused networks, and dangling images.
